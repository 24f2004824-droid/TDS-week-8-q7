# employee_analysis_save_html.py
"""
Employee performance analysis:
- Loads employees.csv
- Prints frequency count for "Marketing"
- Creates a department distribution histogram (bar chart)
- Saves an interactive HTML with the plot and a combined HTML that shows code + plot.
"""

import pandas as pd
import plotly.express as px
import plotly.io as pio
import os
import textwrap

CSV_PATH = "employees.csv"  # must be in same folder
PLOT_HTML = "department_histogram.html"
COMBINED_HTML = "combined_output.html"

# 1) Load data
if not os.path.exists(CSV_PATH):
    raise FileNotFoundError(f"CSV file not found: {CSV_PATH}. Place employees.csv in this folder.")

df = pd.read_csv(CSV_PATH)

# 2) Frequency count for departments
dept_counts = df['department'].value_counts().sort_values(ascending=False)

# 3) Print frequency count for "Marketing" department to console
marketing_count = int(dept_counts.get('Marketing', 0))
print("Frequency count for department 'Marketing':", marketing_count)

# Also print full counts (helpful)
print("\nFull department counts:")
print(dept_counts.to_string())

# 4) Create histogram / bar chart of department distribution using plotly
# We'll make a bar chart (departments on x, counts on y)
dept_df = dept_counts.reset_index()
dept_df.columns = ['department', 'count']

fig = px.bar(
    dept_df,
    x='department',
    y='count',
    title='Employee Count by Department',
    labels={'count': 'Number of Employees', 'department': 'Department'}
)
fig.update_layout(xaxis_tickangle=-45, template='plotly_white')

# 5) Save interactive plot as standalone HTML
# This HTML will be self-contained (Plotly JS included by default)
fig.write_html(PLOT_HTML, include_plotlyjs='cdn', full_html=True)
print(f"\nSaved interactive chart to: {PLOT_HTML}")

# 6) Prepare combined HTML that shows the Python code and embeds the plot
# Read the plot HTML content and the code text, and produce a single HTML file.
with open(PLOT_HTML, 'r', encoding='utf-8') as f:
    plot_html_content = f.read()

# Read this script's source so we can show code inside the HTML
with open(__file__, 'r', encoding='utf-8') as f:
    code_text = f.read()

# Build a simple HTML page that contains:
# - a heading
# - a collapsible code box showing the Python source
# - the plotly HTML (we will embed the plot div/script)
combined_html = f"""
<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8" />
  <title>Employee Department Distribution  Combined</title>
  <meta name="viewport" content="width=device-width, initial-scale=1" />
  <style>
    body {{ font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, "Helvetica Neue", Arial; margin: 20px; }}
    pre.codebox {{ background:#f7f7f8; border:1px solid #ddd; padding:12px; overflow:auto; max-height:400px; }}
    details {{ margin-bottom: 1rem; }}
    h1 {{ font-size: 22px; }}
    .note {{ color: #444; margin-bottom: 12px; }}
  </style>
</head>
<body>
  <h1>Employee Department Distribution</h1>
  <p class="note">This page contains the interactive plot and the Python analysis code that generated it. Email for verification: <strong>24f2004824@ds.study.iitm.ac.in</strong></p>

  <details open>
    <summary><strong>Python analysis script (click to expand)</strong></summary>
    <pre class="codebox">{textwrap.escape(code_text)}</pre>
  </details>

  <h2>Interactive chart</h2>
  <!-- Plotly plot -->
  {plot_html_content}
</body>
</html>
"""

# Write the combined HTML file
with open(COMBINED_HTML, 'w', encoding='utf-8') as f:
    f.write(combined_html)

print(f"Saved combined HTML (code + plot) to: {COMBINED_HTML}")

# End
